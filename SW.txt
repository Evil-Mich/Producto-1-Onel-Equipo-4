SW
enable
configure terminal
no logging console
hostname SW

! --- Configuración Básica, Usuario y SSH ---
ip domain-name itla.local
username equipo4 privilege 15 secret equipo4
enable secret equipo4
crypto key generate rsa modulus 1024

line con 0
 logging synchronous
exit
line vty 0 4
 login local
 transport input ssh
exit

! --- Spanning Tree ---
spanning-tree mode rapid-pvst

! --- Interfaces de Acceso ---
interface range e0/0 - 1
 switchport mode access
 no shutdown
exit

! --- Apagar puertos sin uso por seguridad ---
interface range e0/2 - 3, e1/0 - 3
 shutdown
exit

logging console
do write
end
